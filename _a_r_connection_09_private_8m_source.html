00055 - (NSDictionary *)HTTPFormatHeaderForHTTPMethod:(NSString *)HTTPMethod
00056 {
00057     NSString *const HTTPMethods[] =
00058     {
00059         ARHTTPGetMethod,
00060         ARHTTPPutMethod,
00061         ARHTTPPostMethod,
00062         ARHTTPDeleteMethod,
00063         ARHTTPHeadMethod,
00064     };
00065     NSString *const headerNames[] =
00066     {
00067         @"Accept",
00068         @"Content-Type",
00069         @"Content-Type",
00070         @"Accept",
00071         @"Accept",
00072     };
00073     // Is this too ugly? A dictionary could implement the look-up. But that
00074     // requires building a static dictionary initially and does not allow
00075     // optimisation of searching. Using a simple linear look-up speeds up the
00076     // more common request types, i.e. GET requests. There is a cost, that of
00077     // slower look-up for less common types, e.g. HEAD. Is this a reasonable
00078     // trade-off?
00079     NSUInteger index;
00080     for (index = 0; index < ARDimOf(HTTPMethods); index++)
00081     {
00082         if ([HTTPMethod isEqualToString:HTTPMethods[index]])
00083         {
00084             break;
00085         }
00086     }
00087     NSDictionary *formatHeader;
00088     if (index < ARDimOf(HTTPMethods))
00089     {
00090         formatHeader = [NSDictionary dictionaryWithObject:[[self format] MIMEType] forKey:headerNames[index]];
00091     }
00092     else
00093     {
00094         formatHeader = [NSDictionary dictionary];
00095     }
00096     return formatHeader;
00097 }
