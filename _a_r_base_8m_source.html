00219 - (NSString *)prefixWithOptions:(NSDictionary *)options
00220 {
00221     // The following implementation duplicates some of the functionality
00222     // concerning extraction of prefix parameters from the prefix. See the
00223     // -prefixParameters method. Nevertheless, the replace-in-place approach
00224     // makes the string operations more convenient. The implementation does not
00225     // need to cut apart the colon from its parameter word. The regular
00226     // expression identifies the substitution on our behalf, making it easier to
00227     // remove the colon, access the prefix parameter minus its colon and replace
00228     // both; and all at the same time.
00229     if (options == nil)
00230     {
00231         return [self prefixSourceLazily];
00232     }
00233     return [[NSRegularExpression regularExpressionWithPattern:@":(\\w+)" options:0 error:NULL] stringByReplacingMatchesInString:[self prefixSourceLazily] replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
00234         return [[[options objectForKey:[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"]] description] stringByAddingPercentEscapesUsingEncoding:NSUTF8StringEncoding];
