00179 - (NSString *)prefixWithOptions:(NSDictionary *)options
00180 {
00181     // The following implementation duplicates some of the functionality
00182     // concerning extraction of prefix parameters from the prefix. See the
00183     // -prefixParameters method. Nevertheless, the replace-in-place approach
00184     // makes the string operations more convenient. The implementation does not
00185     // need to cut apart the colon from its parameter word. The regular
00186     // expression identifies the substitution on our behalf, making it easier to
00187     // remove the colon, access the prefix parameter minus its colon and replace
00188     // both; and all at the same time.
00189     if (options == nil)
00190     {
00191         return [self prefixSourceLazily];
00192     }
00193     return [[NSRegularExpression regularExpressionWithPattern:@":(\\w+)" options:0 error:NULL] replaceMatchesInString:[self prefixSourceLazily] replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
00194         return [[[options objectForKey:[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"]] description] stringByAddingPercentEscapesUsingEncoding:NSUTF8StringEncoding];
00195     }];
