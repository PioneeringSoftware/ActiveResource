00164 - (void)mergeAttributes:(NSDictionary *)attributes
00165 {
00166     // Take care not to use -[NSObject setValuesForKeysWithDictionary:] because
00167     // it filters out all those attributes with null values. By design,
00168     // resources preserve all attributes, even those with no values.
00169     for (NSString *attributeName in attributes)
00170     {
00171         id attributeValue = [attributes objectForKey:attributeName];
00172         [_attributes setObject:attributeValue forKey:attributeName];
