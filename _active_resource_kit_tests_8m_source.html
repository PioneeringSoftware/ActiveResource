00069 - (void)setUp
00070 {
00071     post = [[Post alloc] initWithSite:[NSURL URLWithString:@"https://localhost:3000"]];
00072     postComment = [[PostComment alloc] initWithSite:[NSURL URLWithString:@"https://localhost:3000/posts/:post_id"]];
00073     
00074     // You cannot use Comment as a class name. The CarbonCore framework (a
00075     // CoreServices sub-framework) steals this symbol first. Apple has already
00076     // polluted the namespace with a Comment type definition. So we will call it
00077     // PostComment instead, meaning a comment on a post within a blog. However,
00078     // by default, ARBase will translate PostComment to post_comment when
00079     // constructing the resource paths. There is an incongruence between
00080     // Objective-C and Rails, unless Rails uses the same name of course. But the
00081     // comments are just comments in the imaginary Rails application. Hence we
00082     // need now to override the element name. This makes use of the lazy-getter
00083     // approach to configuration. Setting it now overrides the default.
00084     [postComment setElementName:@"comment"];
00085 }

00113 {
00114     // Start off with a very simple test. Just set up a resource. Load it with
00115     // its site URL. Then ensure that the site property responds as it should,
00116     // i.e. as an URL. This test exercises NSURL more than
00117     // ARActiveResource. Never mind.
00118     //
00119     // Make sure that the site URL can accept “prefix parameters,” as Rails dubs

00121     // followed by a regular-expression word. ActiveResourceKit substitutes

00123     ARBase *resource = [[[ARBase alloc] init] autorelease];
00124     [resource setSite:[NSURL URLWithString:@"http://user:password@localhost:3000/resources/:resource_id?x=y;a=b"]];

00126     STAssertEqualObjects([[resource site] user], @"user", nil);

00143 - (void)testPrefixSource
00144 {
00145     // Running the following piece of Ruby:
00146     //
00147     //  require 'active_resource'
00148     //  
00149     //  class Resource < ActiveResource::Base
00150     //    self.prefix = '/resources/:resource_id'
00151     //  end
00152     //  
00153     //  p Resource.prefix(:resource_id => 1)
00154     //
00155     // gives you the following output:
00156     //
00157     //  "/resources/1"
00158     //
00159     // The following test performs the same thing but using Objective-C.
00160     //
00161     // Note that the options can contain numbers and other objects. Method
00162     // -[ARActiveResource prefixWithOptions:] places the “description” of the
00163     // object answering to the prefix-parameter key (resource_id in this test
00164     // case). Hence the options dictionary can contain various types answering
00165     // to -[NSObject description], not just strings.
00166     ARBase *base = [[[ARBase alloc] init] autorelease];
00167     [base setPrefixSource:@"/resources/:resource_id"];
00168     NSDictionary *options = [NSDictionary dictionaryWithObject:[NSNumber numberWithInt:1] forKey:@"resource_id"];
00169     NSString *prefix = [base prefixWithOptions:options];
00170     STAssertEqualObjects(prefix, @"/resources/1", nil);
00171 }

00235     // scheme pre-action comprises a shell script containing:

00238     //  [ -f tmp/pids/server.pid ] || "$HOME/.rvm/bin/rvm-shell" -c "rake db:setup db:fixtures:load ; rails s -d -P `pwd`/tmp/pids/server-xcode.pid"
00239     //
00240     // The script launches a Rails server in the background using RVM, if and
00241     // only if the default server is not already running. It assumes that RVM is
00242     // installed and carries the necessary Ruby gems. The post-action for the
00243     // Test scheme contains the following shell script. It sends an interrupt
00244     // signal to the server to shut it down.
00245     //
00246     //  cd "$SRCROOT/active-resource-kit-tests"

00250         STAssertNotNil(resource, nil);
