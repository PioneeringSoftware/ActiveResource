00063 - (void)setUp
00064 {
00065     post = [[Post alloc] initWithSite:ActiveResourceKitTestsBaseURL()];
00066     postComment = [[PostComment alloc] initWithSite:[NSURL URLWithString:@"/posts/:post_id" relativeToURL:ActiveResourceKitTestsBaseURL()]];
00067     
00068     // You cannot use Comment as a class name. The CarbonCore framework (a
00069     // CoreServices sub-framework) steals this symbol first. Apple has already
00070     // polluted the namespace with a Comment type definition. So we will call it
00071     // PostComment instead, meaning a comment on a post within a blog. However,
00072     // by default, ARService will translate PostComment to post_comment when
00073     // constructing the resource paths. There is an incongruence between
00074     // Objective-C and Rails, unless Rails uses the same name of course. But the
00075     // comments are just comments in the imaginary Rails application. Hence we
00076     // need now to override the element name. This makes use of the lazy-getter
00077     // approach to configuration. Setting it now overrides the default.
00078     [postComment setElementName:@"comment"];
00079 }

00100 - (void)testSetUpSite
00101 {
00102     // Start off with a very simple test. Just set up a resource. Load it with
00103     // its site URL. Then ensure that the site property responds as it should,
00104     // i.e. as an URL. This test exercises NSURL more than
00105     // ARActiveResource. Never mind.
00106     //
00107     // Make sure that the site URL can accept “prefix parameters,” as Rails dubs
00108     // them. This term refers to path elements beginning with a colon and
00109     // followed by a regular-expression word. ActiveResourceKit substitutes
00110     // these for actual parameters.
00111     ARService *service = [[ARService alloc] init];
00112     [service setSite:[NSURL URLWithString:@"http://user:password@localhost:3000/resources/:resource_id?x=y;a=b"]];
00113     STAssertEqualObjects([[service site] scheme], @"http", nil);
00114     STAssertEqualObjects([[service site] user], @"user", nil);
00115     STAssertEqualObjects([[service site] password], @"password", nil);
00116     STAssertEqualObjects([[service site] host], @"localhost", nil);
00117     STAssertEqualObjects([[service site] port], [NSNumber numberWithInt:3000], nil);
00118     STAssertEqualObjects([[service site] path], @"/resources/:resource_id", nil);
00119     STAssertEqualObjects([[service site] query], @"x=y;a=b", nil);

00131 - (void)testPrefixSource
00132 {
00133     // Running the following piece of Ruby:
00134     //
00135     //  require 'active_resource'
00136     //  
00137     //  class Resource < ActiveResource::Base
00138     //    self.prefix = '/resources/:resource_id'
00139     //  end
00140     //  
00141     //  p Resource.prefix(:resource_id => 1)
00142     //
00143     // gives you the following output:
00144     //
00145     //  "/resources/1"
00146     //
00147     // The following test performs the same thing but using Objective-C.
00148     //
00149     // Note that the options can contain numbers and other objects. Method
00150     // -[ARActiveResource prefixWithOptions:] places the “description” of the
00151     // object answering to the prefix-parameter key (resource_id in this test
00152     // case). Hence the options dictionary can contain various types answering
00153     // to -[NSObject description], not just strings.
00154     ARService *service = [[ARService alloc] init];
00155     [service setPrefixSource:@"/resources/:resource_id"];
00156     NSDictionary *options = [NSDictionary dictionaryWithObject:[NSNumber numberWithInt:1] forKey:@"resource_id"];
00157     NSString *prefix = [service prefixWithOptions:options];
00158     STAssertEqualObjects(prefix, @"/resources/1", nil);
00159 }
