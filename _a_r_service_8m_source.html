00250 - (NSString *)prefixWithOptions:(NSDictionary *)options
00251 {
00252     // The following implementation duplicates some of the functionality
00253     // concerning extraction of prefix parameters from the prefix. See the
00254     // -prefixParameters method. Nevertheless, the replace-in-place approach
00255     // makes the string operations more convenient. The implementation does not
00256     // need to cut apart the colon from its parameter word. The regular
00257     // expression identifies the substitution on our behalf, making it easier to
00258     // remove the colon, access the prefix parameter minus its colon and replace
00259     // both; and all at the same time.
00260     if (options == nil)
00261     {
00262         return [self prefixSourceLazily];
00263     }
