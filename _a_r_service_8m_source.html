250 - (NSString *)prefixWithOptions:(NSDictionary *)options
251 {
252  // The following implementation duplicates some of the functionality
253  // concerning extraction of prefix parameters from the prefix. See the
254  // -prefixParameters method. Nevertheless, the replace-in-place approach
255  // makes the string operations more convenient. The implementation does not
256  // need to cut apart the colon from its parameter word. The regular
257  // expression identifies the substitution on our behalf, making it easier to
258  // remove the colon, access the prefix parameter minus its colon and replace
259  // both; and all at the same time.
260  if (options == nil)
261  {
262  return [self prefixSourceLazily];
263  }
264  return [[NSRegularExpression regularExpressionWithPattern:@":(\\w+)" options:0 error:NULL] stringByReplacingMatchesInString:[self prefixSourceLazily] replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
265  return [[[options objectForKey:[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"]] description] stringByAddingPercentEscapesUsingEncoding:NSUTF8StringEncoding];
266  }];
267 }
