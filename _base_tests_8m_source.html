00058     // Do not execute a run loop with nil IDs. The answer (exists equal to NO)
00059     // comes back immediately. The completion block runs before the
00060     // exists-with-ID method returns.

00062         STAssertFalse(exists, nil);

00065     [peopleService existsWithID:[NSNumber numberWithInt:1] options:nil completionHandler:^(ARHTTPResponse *response, BOOL exists, NSError *error) {
00066         STAssertTrue(exists, nil);
00067         [self setStop:YES];
00068     }];
00069     [self runUntilStop];
00070     
00071     [peopleService existsWithID:[NSNumber numberWithInt:99] options:nil completionHandler:^(ARHTTPResponse *response, BOOL exists, NSError *error) {
