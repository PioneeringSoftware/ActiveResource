62 - (NSString *)defaultPrefixSource
63 {
64  // Asking for the site path answers only the non-base path as the path, even
65  // if the URL comprises relative path components based on a base URL which
66  // itself includes a path. The base URL's path disappears from the
67  // equation. Is this a bug in Foundation? Work around it by computing the
68  // prefix source through concatenation of the two paths.
69  NSURL *site = [self site];
70  NSURL *baseURL = [site baseURL];
71  NSString *path = [site path];
72  return baseURL ? [[baseURL path] stringByAppendingPathComponent:path] : path;
73 }
