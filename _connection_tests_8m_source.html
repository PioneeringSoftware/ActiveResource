00032 - (void)testErrorForResponse
00033 {
00034     NSError *(^errorForResponse)(NSInteger code) = ^(NSInteger code) {
00035         return [ARConnection errorForResponse:[[ARHTTPResponse alloc] initWithHTTPURLResponse:[[NSHTTPURLResponse alloc] initWithURL:ActiveResourceKitTestsBaseURL() statusCode:code HTTPVersion:@"HTTP/1.1" headerFields:nil] body:nil]];
00036     };
00037     // valid responses: 2xx and 3xx
00038     for (NSNumber *code in [NSArray arrayWithObjects:[NSNumber numberWithInt:200], [NSNumber numberWithInt:299], [NSNumber numberWithInt:300], [NSNumber numberWithInt:399], nil])
00039     {
00040         STAssertNil(errorForResponse([code integerValue]), nil);
00041     }
00042     // redirection
00043     NSArray *redirectCodes = [NSArray arrayWithObjects:[NSNumber numberWithInt:301], [NSNumber numberWithInt:302], [NSNumber numberWithInt:303], [NSNumber numberWithInt:307], nil];
00044     NSArray *redirectDescriptions = [NSArray arrayWithObjects:@"moved permanently", @"found", @"see other", @"temporarily redirected", nil];
00045     NSDictionary *redirectDescriptionForCode = [NSDictionary dictionaryWithObjects:redirectDescriptions forKeys:redirectCodes];
00046     for (NSNumber *code in redirectCodes)
00047     {
00048         NSError *error = errorForResponse([code integerValue]);
00049         STAssertNotNil(error, nil);
00050         STAssertEquals([error code], (NSInteger)ARRedirectionErrorCode, nil);
00051         STAssertEqualObjects([[error userInfo] objectForKey:NSLocalizedDescriptionKey], [redirectDescriptionForCode objectForKey:code], nil);
00052     }
00053     // client errors: 4xx
00054     struct
00055     {
00056         NSInteger statusCode;
00057         NSInteger errorCode;
00058     }
00059     clientCodesAndErrors[] =
00060     {
00061         { 400, (NSInteger)ARBadRequestErrorCode },
00062         { 401, (NSInteger)ARUnauthorizedAccessErrorCode },
00063         { 403, (NSInteger)ARForbiddenAccessErrorCode },
00064         { 404, (NSInteger)ARResourceNotFoundErrorCode },
00065         { 405, (NSInteger)ARMethodNotAllowedErrorCode },
00066         { 409, (NSInteger)ARResourceConflictErrorCode },
00067         { 410, (NSInteger)ARResourceGoneErrorCode },
00068         { 422, (NSInteger)ARResourceInvalidErrorCode },
00069     };
00070     for (NSUInteger i = 0; i < ASDimOf(clientCodesAndErrors); i++)
00071     {
00072         STAssertEquals([errorForResponse(clientCodesAndErrors[i].statusCode) code], clientCodesAndErrors[i].errorCode, nil);
00073     }
00074     for (NSInteger statusCode = 402; statusCode <= 499; statusCode++)
00075     {
00076         NSUInteger i;
00077         for (i = 0; i < ASDimOf(clientCodesAndErrors) && statusCode != clientCodesAndErrors[i].statusCode; i++);
00078         if (i == ASDimOf(clientCodesAndErrors))
00079         {
00080             STAssertEquals([errorForResponse(statusCode) code], (NSInteger)ARClientErrorCode, nil);
00081         }
00082     }
00083     // server errors: 5xx
00084     for (NSInteger statusCode = 500; statusCode <= 599; statusCode++)
00085     {
00086         STAssertEquals([errorForResponse(statusCode) code], (NSInteger)ARServerErrorCode, nil);
00087     }
00088 }
