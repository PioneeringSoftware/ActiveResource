00183 - (NSString *)prefixWithOptions:(NSDictionary *)options
00184 {
00185     // The following implementation duplicates some of the functionality
00186     // concerning extraction of prefix parameters from the prefix. See the
00187     // -prefixParameters method. Nevertheless, the replace-in-place approach
00188     // makes the string operations more convenient. The implementation does not
00189     // need to cut apart the colon from its parameter word. The regular
00190     // expression identifies the substitution on our behalf, making it easier to
00191     // remove the colon, access the prefix parameter minus its colon and replace
00192     // both; and all at the same time.
00193     if (options == nil)
00194     {
00195         return [self prefixSourceOrDefault];
00196     }
00197     return [[NSRegularExpression regularExpressionWithPattern:@":(\\w+)" options:0 error:NULL] replaceMatchesInString:[self prefixSourceOrDefault] replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
00198         return [[[options objectForKey:[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"]] description] stringByAddingPercentEscapesUsingEncoding:NSUTF8StringEncoding];
00199     }];
