00071 - (void)loadAttributes:(NSDictionary *)attributes removeRoot:(BOOL)removeRoot
00072 {
00073     NSDictionary *prefixOptions = nil;
00074     [[self base] splitOptions:attributes prefixOptions:&prefixOptions queryOptions:&attributes];
00075     if ([attributes count] == 1)
00076     {
00077         removeRoot = [[[self base] elementName] isEqualToString:[[[attributes allKeys] objectAtIndex:0] description]];
00078     }
00079     if (removeRoot)
00080     {
00081         attributes = ARRemoveRoot(attributes);

00083     [self setAttributes:attributes];
